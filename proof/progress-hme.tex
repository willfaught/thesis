$e_{M}^{1}=\overline{n}$ by canonical forms (Lemma \ref{cf}).  $^{N}MH^{N}\;\overline{n}\rightarrow\overline{n}$.
\end{subcase}
\begin{subcase}

\begin{subcase}

\begin{subcase}

\end{case}
