\begin{lemma}
\label{tms}
If $\Gamma,x:T_{1}\vdash_{HM}e_{HM}:T_{2}$ and $\Gamma\vdash_{HM}y:T_{1}$ then $\Gamma\vdash_{HM}e_{HM}[y/x]:T_{2}$.  If $\Gamma,x:TST\vdash_{S}e_{S}:TST$ and $\Gamma\vdash_{S}y:TST$ then $\Gamma\vdash_{S}e_{S}[y/x]:TST$.
\begin{proof}
By structural induction.
\end{proof}
\end{lemma}
